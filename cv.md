function digitalRoot(n) {
  // Base case: if n is a single-digit number, return n
  if (n < 10) {
    return n;
  } else {
    // Calculate the sum of the digits
    const sum = (n + '').split('').reduce((acc, cur) => acc + +cur, 0);
    // Recursive call with the new sum
    return digitalRoot(sum);
  }
}
